\newcommand\bigfont[1]{{\fontsize{50}{60}\sffamily\bfseries\selectfont #1}}

\newdimen\cardwidth
\cardwidth=60mm
\newdimen\cardheight
\cardheight=90mm
\newdimen\cardborder
\cardborder=5mm

\newcount\auxnum
\newenvironment{CardFront}{%
  \begin{tikzpicture}
    \draw[rounded corners=5mm] (0,0) rectangle (\cardwidth,\cardheight) coordinate (URex);
    \path (0,0)--(\cardwidth,\cardheight) coordinate[midway] (Center);
    \path (URex)--+(-\cardborder,-\cardborder) coordinate (UR);
    \fill[green!20, rounded corners=\cardborder] (\cardborder,\cardborder) coordinate (LL) rectangle (UR);
}{
  \end{tikzpicture}%
}
\colorlet{nullcolor}{yellow!30}
\colorlet{middlecolor}{green!30}
\colorlet{highcolor}{blue!30}
\newcommand\CardFrontNumbered[1]{%
  \begin{CardFront}
    \ifnum#1<8
      \auxnum=#1
      \multiply\auxnum by 12
      \colorlet{numcolor}{middlecolor!\the\auxnum!nullcolor}
      \def\eight{fill color=white}
    \else
      \auxnum=#1
      \advance\auxnum by -8
      \multiply\auxnum by 12
      \colorlet{numcolor}{highcolor!\the\auxnum!middlecolor}
      \def\eight{fill color=black}
    \fi
    \node[draw, line width=1.5pt, fill=numcolor, circle, anchor=center, minimum size=0.5*\cardwidth] at (Center) {\bigfont{#1}};
    \path (Center)++(-0.21*\cardwidth,-0.35*\cardwidth) circle[draw,fill,\eight, radius=3mm];
    \path (Center)++(-0.07*\cardwidth,-0.35*\cardwidth) circle[draw,fill,\eight, radius=3mm];
    \path (Center)++(+0.07*\cardwidth,-0.35*\cardwidth) circle[draw,fill,\eight, radius=3mm];
    \path (Center)++(+0.21*\cardwidth,-0.35*\cardwidth) circle[draw,fill,\eight, radius=3mm];
  \end{CardFront}%
}
